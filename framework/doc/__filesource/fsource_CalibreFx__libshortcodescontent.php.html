  24. /**
  25.  * Produces the date of post publication.
  26.  *
  27.  * Supported shortcode attributes are:
  28.  *   after (output after link, default is empty string),
  29.  *   before (output before link, default is empty string),
  30.  *   format (date format, default is value in date_format option field),
  31.  *   label (text following 'before' output, but before date).
  32.  *
  33.  */
  34. function calibrefx_post_date_shortcode( $atts ) {
  35.  
  36.     $defaults = array(
  37.         'after'  => '',
  38.         'before' => '',
  39.         'format' => get_option( 'date_format' ),
  40.         'label'  => '',
  41.     );
  42.     $atts = shortcode_atts( $defaults, $atts );
  43.  
  44.     $display = ( 'relative' == $atts['format'] ) ? calibrefx_human_time_diff( get_the_time( 'U' ), current_time( 'timestamp' ) ) . ' ' . __( 'ago', 'calibrefx' ) : get_the_time( $atts['format'] );
  45.  
  46.     $output = sprintf( '<span class="date published time" title="%5$s">%1$s%3$s%4$s%2$s</span> ', $atts['before'], $atts['after'], $atts['label'], $display, get_the_time( 'c' ) );
  47.  
  48.     return apply_filters( 'calibrefx_post_date_shortcode', $output, $atts );
  49.  
  50. }

  153. /**
  154.  * Produces the link to the current post comments.
  155.  *
  156.  * Supported shortcode attributes are:
  157.  *   after (output after link, default is empty string),
  158.  *   before (output before link, default is empty string),
  159.  *   hide_if_off (hide link if comments are off, default is 'enabled' (true)),
  160.  *   more (text when there is more than 1 comment, use % character as placeholder
  161.  *     for actual number, default is '% Comments')
  162.  *   one (text when there is exactly one comment, default is '1 Comment'),
  163.  *   zero (text when there are no comments, default is 'Leave a Comment').
  164.  *
  165.  */
  166. function calibrefx_post_comments_shortcode( $atts ) {
  167.  
  168.     $defaults = array(
  169.         'after'       => '',
  170.         'before'      => ' &#8226; ',
  171.         'hide_if_off' => 'enabled',
  172.         'more'        => __( '% Comments', 'calibrefx' ),
  173.         'one'         => __( '1 Comment', 'calibrefx' ),
  174.         'zero'        => __( 'Leave a Comment', 'calibrefx' ),
  175.     );
  176.     $atts = shortcode_atts( $defaults, $atts );
  177.  
  178.     if ( ( ! calibrefx_get_option( 'comments_posts' ) || ! comments_open() ) && 'enabled' === $atts['hide_if_off'] )
  179.         return;
  180.  
  181.     // Darn you, WordPress!
  182.     ob_start();
  183.     comments_number( $atts['zero'], $atts['one'], $atts['more'] );
  184.     $comments = ob_get_clean();
  185.  
  186.     $comments = sprintf( '<a href="%s">%s</a>', get_comments_link(), $comments );
  187.  
  188.     $output = sprintf( '<span class="post-comments">%2$s%1$s%3$s</span>', $comments, $atts['before'], $atts['after'] );
  189.  
  190.     return apply_filters( 'calibrefx_post_comments_shortcode', $output, $atts );
  191.  
  192. }
