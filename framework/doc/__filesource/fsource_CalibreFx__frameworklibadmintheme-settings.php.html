  25.  * This function will save or reset settings
  26.  */
  27. function calibrefx_register_theme_settings() {
  28.     register_setting( CALIBREFX_SETTINGS_FIELD, CALIBREFX_SETTINGS_FIELD );
  29.     add_option( CALIBREFX_SETTINGS_FIELD, calibrefx_theme_settings_defaults() );
  30.  
  31.     if ( ! isset($_REQUEST['page']) || $_REQUEST['page'] != 'calibrefx' )
  32.         return;
  33.  
  34.     if ( calibrefx_get_option('reset') ) {
  35.         update_option(CALIBREFX_SETTINGS_FIELD, calibrefx_theme_settings_defaults());
  36.  
  37.         calibrefx_admin_redirect( 'calibrefx', array( 'reset' => 'true' ) );
  38.         exit;
  39.     }
  40. }

  46. function calibrefx_theme_settings_notice() {
  47.  
  48.     if ( ! isset( $_REQUEST['page'] ) || $_REQUEST['page'] != 'calibrefx' )
  49.         return;
  50.  
  51.     if ( isset( $_REQUEST['reset'] ) && 'true' == $_REQUEST['reset'] )
  52.         echo '<div id="message" class="updated"><p><strong>' . __( 'Settings reset.', 'calibrefx' ) . '</strong></p></div>';
  53.     elseif ( isset( $_REQUEST['settings-updated'] ) && $_REQUEST['settings-updated'] == 'true' )
  54.         echo '<div id="message" class="updated"><p><strong>' . __( 'Settings saved.' ) . '</strong></p></div>';
  55.
