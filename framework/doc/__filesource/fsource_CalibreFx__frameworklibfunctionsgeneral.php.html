  36. function calibrefx_admin_redirect( $page, $query_args = array() ) {
  37.  
  38.     if ( ! $page )
  39.         return;

  41.     $url = menu_page_url( $page, false );

  45.             $url = add_query_arg( $key, $value, $url );
  46.         }

  48.

  53. /**
  54.  * Display author box and its contents.
  55.  */
  56. function calibrefx_author_box( $context = '' ) {
  57.  
  58.     global $authordata;
  59.     
  60.     $authordata    = is_object( $authordata ) ? $authordata : get_userdata( get_query_var( 'author' ) );
  61.     $gravatar_size = apply_filters( 'calibrefx_author_box_gravatar_size', 70, $context );
  62.     $gravatar      = get_avatar( get_the_author_meta( 'email' ), $gravatar_size );
  63.     $title         = apply_filters( 'calibrefx_author_box_title', sprintf( '<strong>%s %s</strong>', __( 'About', 'genesis' ), get_the_author() ), $context );
  64.     $description   = wpautop( get_the_author_meta( 'description' ) );
  65.  
  66.     /** The author box markup, contextual */
  67.     $pattern = $context == 'single' ? '<div class="author-box well"><div>%s %s<br />%s</div></div><!-- end .authorbox-->' : '<div class="author-box well">%s<h1>%s</h1><div>%s</div></div><!-- end .authorbox-->';
  68.     
  69.     
  70.     echo apply_filters( 'calibrefx_author_box', sprintf( $pattern, $gravatar, $title, $description ), $context, $pattern, $gravatar, $title, $description );
  71.  
  72. }
