  23. /**
  24.  * Return a phrase shortened in length to a maximum number of characters.
  25.  *
  26.  * Result will be truncated at the last white space in the original
  27.  * string. In this function the word separator is a single space (' ').
  28.  * Other white space characters (like newlines and tabs) are ignored.
  29.  *
  30.  * If the first $max_characters of the string do contain a space
  31.  * character, an empty string will be returned.
  32.  *
  33.  * @param string $phrase A string to be shortened.
  34.  * @param integer $max_characters The maximum number of characters to return.
  35.  * @return string 
  36.  */
  37. function calibrefx_truncate_phrase($phrase, $max_characters) {
  38.  
  39.     $phrase = trim( $phrase );
  40.  
  41.     if ( strlen($phrase) > $max_characters ) {
  42.  
  43.         // Truncate $phrase to $max_characters + 1
  44.         $phrase = substr($phrase, 0, $max_characters + 1);
  45.  
  46.         // Truncate to the last space in the truncated string.
  47.         $phrase = trim(substr($phrase, 0, strrpos($phrase, ' ')));
  48.     }
  49.  
  50.     return $phrase;
  51. }
  52.  
  53. /**
  54.  * This function strips out tags and shortcodes,
  55.  * limits the output to $max_char characters,
  56.  * and appends an ellipses and more link to the end.
  57.  */
  58. function get_the_content_limit($max_char, $more_link_text = '(more...)', $stripteaser = 0) {
  59.  
  60.     $content = get_the_content('', $stripteaser);
  61.  
  62.     // Strip tags and shortcodes
  63.     $content = strip_tags(strip_shortcodes($content), apply_filters('get_the_content_limit_allowedtags', '<script>,<style>'));
  64.  
  65.     // Inline styles/scripts
  66.     $content = trim(preg_replace('#<(s(cript|tyle)).*?</\1>#si', '', $content));
  67.  
  68.     // Truncate $content to $max_char
  69.     $content = calibrefx_truncate_phrase($content, $max_char);
  70.  
  71.     // More Link?
  72.     if ( $more_link_text ) {
  73.         $link = apply_filters( 'get_the_content_more_link', sprintf( '%s <a href="%s" class="more-link">%s</a>', '&hellip;', get_permalink(), $more_link_text ), $more_link_text );
  74.  
  75.         $output = sprintf('<p>%s %s</p>', $content, $link);
  76.     }
  77.     else {
  78.         $output = sprintf('<p>%s</p>', $content);
  79.     }
  80.  
  81.     return apply_filters('get_the_content_limit', $output, $content, $link, $max_char);
  82.  
  83. }

  95. /**
  96.  * Calculate the time difference - a replacement for human_time_diff() until it
  97.  * is improved.
  98.  *
  99.  * Based on BuddyPress function bp_core_time_since(), which in turn is based on
  100.  * functions created by Dunstan Orchard - http://1976design.com
  101.  *
  102.  * This function will return an text representation of the time elapsed since a
  103.  * given date, giving the two largest units e.g.:
  104.  *  - 2 hours and 50 minutes
  105.  *  - 4 days
  106.  *  - 4 weeks and 6 days
  107.  *
  108.  * @since 1.7.0
  109.  *
  110.  * @param $older_date int Unix timestamp of date you want to calculate the time since for
  111.  * @param $newer_date int Unix timestamp of date to compare older date to. Default false (current time)
  112.  * @return str The time difference
  113.  */
  114. function calibrefx_human_time_diff( $older_date, $newer_date = false ) {
  115.  
  116.     /** If no newer date is given, assume now */
  117.     $newer_date = $newer_date ? $newer_date : time();
  118.  
  119.     /** Difference in seconds */
  120.     $since = absint( $newer_date - $older_date );
  121.  
  122.     if ( ! $since )
  123.         return '0 ' . _x( 'seconds', 'time difference', 'calibrefx' );
  124.  
  125.     /** Hold units of time in seconds, and their pluralised strings (not translated yet) */
  126.     $units = array(
  127.         array( 31536000, _nx_noop( '%s year', '%s years', 'time difference' ) ),  // 60 * 60 * 24 * 365
  128.         array( 2592000, _nx_noop( '%s month', '%s months', 'time difference' ) ), // 60 * 60 * 24 * 30
  129.         array( 604800, _nx_noop( '%s week', '%s weeks', 'time difference' ) ),    // 60 * 60 * 24 * 7
  130.         array( 86400, _nx_noop( '%s day', '%s days', 'time difference' ) ),       // 60 * 60 * 24
  131.         array( 3600, _nx_noop( '%s hour', '%s hours', 'time difference' ) ),      // 60 * 60
  132.         array( 60, _nx_noop( '%s minute', '%s minutes', 'time difference' ) ),
  133.         array( 1, _nx_noop( '%s second', '%s seconds', 'time difference' ) ),
  134.     );
  135.  
  136.     /** Step one: the first unit */
  137.     for ( $i = 0, $j = count( $units ); $i < $j; $i++ ) {
  138.         $seconds = $units[$i][0];
  139.  
  140.         /** Finding the biggest chunk (if the chunk fits, break) */
  141.         if ( ( $count = floor( $since / $seconds ) ) != 0 )
  142.             break;
  143.     }
  144.  
  145.     /** Translate unit string, and add to the output */
  146.     $output = sprintf( translate_nooped_plural( $units[$i][1], $count, 'calibrefx' ), $count );
  147.  
  148.     /** Note the next unit */
  149.     $ii = $i + 1;
  150.  
  151.     /** Step two: the second unit */
  152.     if ( $ii <= $j ) {
  153.         $seconds2 = $units[$ii][0];
  154.  
  155.         /** Check if this second unit has a value > 0 */
  156.         if ( ( $count2 = floor( ( $since - ( $seconds * $count ) ) / $seconds2 ) ) != 0 )
  157.             /** Add translated separator string, and translated unit string */
  158.             $output .= sprintf( ' %s ' . translate_nooped_plural( $units[$ii][1], $count2, 'calibrefx' ),    _x( 'and', 'separator in time difference', 'calibrefx' ),    $count2    );
  159.     }
  160.  
  161.     return $output;
  162. }

  196. /**
  197.  * Adds links to the contents of a tweet.
  198.  */
  199. function calibrefx_tweet_linkify( $text ) {
  200.  
  201.     $text = preg_replace( "#(^|[\n ])([\w]+?://[\w]+[^ \"\n\r\t< ]*)#", '\\1<a href="\\2" target="_blank">\\2</a>', $text );
  202.     $text = preg_replace( "#(^|[\n ])((www|ftp)\.[^ \"\t\n\r< ]*)#", '\\1<a href="http://\\2" target="_blank">\\2</a>', $text );
  203.     $text = preg_replace( '/@(\w+)/', '<a href="http://www.twitter.com/\\1" target="_blank">@\\1</a>', $text );
  204.     $text = preg_replace( '/#(\w+)/', '<a href="http://search.twitter.com/search?q=\\1" target="_blank">#\\1</a>', $text );
  205.  
  206.     return $text;
  207.  
  208. }
