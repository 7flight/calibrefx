  290. function calibrefx_layout_selector( $args = array() ) {
  291.  
  292.     /** Merge defaults with user args */
  293.     $args = wp_parse_args( $args, array(

  306.                 esc_attr( $data['label'] ),
  307.                 esc_attr( $class ),
  308.                 esc_url( $data['img'] ),
  309.                 esc_attr( $args['name'] ),
  310.                 esc_attr( $id ),
  311.                 checked( $id, $args['selected'], false )
  312.         );
  313.  
  314.     }
  315.     
  316.     /** Echo or Return output */
  317.     if ( $args['echo'] )

  320.         return $output;

  322. }
