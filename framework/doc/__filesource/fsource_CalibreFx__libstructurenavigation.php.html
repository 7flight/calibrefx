  24. /**
  25.  * Display Post Navigation
  26.  */
  27. function calibrefx_posts_nav() {
  28.  
  29.     $nav = calibrefx_get_option( 'posts_nav' );
  30.  
  31.     if( 'prev-next' == $nav )
  32.         calibrefx_prev_next_posts_nav();
  33.     elseif( 'numeric' == $nav )
  34.         calibrefx_numeric_posts_nav();
  35.     else
  36.         calibrefx_older_newer_posts_nav();
  37.  
  38. }

  40. /**
  41.  * Echoes post navigation in Older Posts / Newer Posts format.
  42.  */
  43. function calibrefx_older_newer_posts_nav() {
  44.  
  45.     $older_link = get_next_posts_link( apply_filters( 'calibrefx_older_link_text', '&larr; ' ) . __( 'Older Posts', 'calibrefx' ) );
  46.     $newer_link = get_previous_posts_link( apply_filters( 'calibrefx_newer_link_text', __( 'Newer Posts', 'calibrefx' ) . ' &rarr;' ) );
  47.  
  48.     $older = $older_link ? '<div class="previous">' . $older_link . '</div>' : '';
  49.     $newer = $newer_link ? '<div class="next">' . $newer_link . '</div>' : '';
  50.  
  51.     $nav = '<div class="navigation pager">' . $older . $newer . '</div><!-- end .navigation -->';
  52.  
  53.     if ( $older || $newer )
  54.         echo $nav;
  55.  
  56. }

  58. /**
  59.  * Echoes post navigation in Previous Posts / Next Posts format.
  60.  *
  61.  */
  62. function calibrefx_prev_next_posts_nav() {
  63.  
  64.     $prev_link = get_previous_posts_link( apply_filters( 'calibrefx_prev_link_text', '&larr; ' . __( 'Previous Page', 'calibrefx' ) ) );
  65.     $next_link = get_next_posts_link( apply_filters( 'calibrefx_next_link_text', __( 'Next Page', 'calibrefx' ) . ' &rarr;' ) );
  66.  
  67.     $prev = $prev_link ? '<div class="previous">' . $prev_link . '</div>' : '';
  68.     $next = $next_link ? '<div class="next">' . $next_link . '</div>' : '';
  69.  
  70.     $nav = '<div class="navigation pager">' . $prev . $next . '</div><!-- end .navigation -->';
  71.  
  72.     if ( $prev || $next )
  73.         echo $nav;
  74. }

  76. /**
  77.  * Echoes post navigation in page numbers format (similar to WP-PageNavi).
  78.  *
  79.  * The links, if needed, are ordered as:
  80.  *   previous page arrow,
  81.  *   first page,
  82.  *   up to two pages before current page,
  83.  *   current page,
  84.  *   up to two pages after the current page,
  85.  *   last page,
  86.  *   next page arrow.
  87.  */
  88. function calibrefx_numeric_posts_nav() {
  89.  
  90.     if( is_singular() )
  91.         return;
  92.  
  93.     global $wp_query;
  94.  
  95.     /** Stop execution if there's only 1 page */
  96.     if( $wp_query->max_num_pages <= 1 )
  97.         return;
  98.  
  99.     $paged = get_query_var( 'paged' ) ? absint( get_query_var( 'paged' ) ) : 1;
  100.     $max   = intval( $wp_query->max_num_pages );
  101.  
  102.     /**    Add current page to the array */
  103.     if ( $paged >= 1 )
  104.         $links[] = $paged;
  105.  
  106.     /**    Add the pages around the current page to the array */
  107.     if ( $paged >= 3 ) {
  108.         $links[] = $paged - 1;
  109.         $links[] = $paged - 2;
  110.     }
  111.  
  112.     if ( ( $paged + 2 ) <= $max ) {
  113.         $links[] = $paged + 2;
  114.         $links[] = $paged + 1;
  115.     }
  116.  
  117.     echo '<div class="navigation pagination pagination-centered"><ul>' . "\n";
  118.  
  119.     /**    Previous Post Link */
  120.     if ( get_previous_posts_link() )
  121.         printf( '<li>%s</li>' . "\n", get_previous_posts_link( apply_filters( 'calibrefx_prev_link_text', '&laquo; ' . __( 'Previous Page', 'calibrefx' ) ) ) );
  122.  
  123.     /**    Link to first page, plus ellipses if necessary */
  124.     if ( ! in_array( 1, $links ) ) {
  125.         $class = 1 == $paged ? ' class="active"' : '';
  126.  
  127.         printf( '<li%s><a href="%s">%s</a></li>' . "\n", $class, get_pagenum_link( 1 ), '1' );
  128.  
  129.         if ( ! in_array( 2, $links ) )
  130.             echo '<li>&hellip;</li>';
  131.     }
  132.  
  133.     /**    Link to current page, plus 2 pages in either direction if necessary */
  134.     sort( $links );
  135.     foreach ( (array) $links as $link ) {
  136.         $class = $paged == $link ? ' class="active"' : '';
  137.         printf( '<li%s><a href="%s">%s</a></li>' . "\n", $class, get_pagenum_link( $link ), $link );
  138.     }
  139.  
  140.     /**    Link to last page, plus ellipses if necessary */
  141.     if ( ! in_array( $max, $links ) ) {
  142.         if ( ! in_array( $max - 1, $links ) )
  143.             echo '<li>&hellip;</li>' . "\n";
  144.  
  145.         $class = $paged == $max ? ' class="active"' : '';
  146.         printf( '<li%s><a href="%s">%s</a></li>' . "\n", $class, get_pagenum_link( $max ), $max );
  147.     }
  148.  
  149.     /**    Next Post Link */
  150.     if ( get_next_posts_link() )
  151.         printf( '<li>%s</li>' . "\n", get_next_posts_link( apply_filters( 'calibrefx_next_link_text', __( 'Next Page', 'calibrefx' ) . ' &raquo;' ) ) );
  152.  
  153.     echo '</ul></div>' . "\n";
  154.  
  155. }
