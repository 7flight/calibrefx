  23. /**
  24.  * This function calibrefx_update_check is to ...
  25.  */
  26. function calibrefx_update_check() {
  27.     global $wp_version;
  28.     
  29.     /** Get time of last update check */
  30.     $calibrefx_update = get_transient( 'calibrefx-update' );
  31.     //$calibrefx_update = array("new_version" => '1.1', "changelog_url" => '#', "url" => "themes-url", "package"=>"download-url", ""=>"");
  32.     
  33.     /** If it has expired, do an update check */
  34.     if ( ! $calibrefx_update ) {
  35.         $url     = 'http://api.calibrefx.com/themes-update-check';
  36.         $options = apply_filters(
  37.             'calibrefx_update_remote_post_options',
  38.             array(
  39.                 'body' => array(
  40.                     'theme_version'   => FRAMEWORK_VERSION,
  41.                     'wp_version'      => $wp_version,
  42.                     'php_version'     => phpversion(),
  43.                     'uri'             => home_url(),
  44.                     'user-agent'      => "WordPress/$wp_version;",
  45.                 ),
  46.             )
  47.         );
  48.  
  49.         $response = wp_remote_post( $url, $options );
  50.         $calibrefx_update = wp_remote_retrieve_body( $response );
  51.  
  52.         /** If an error occurred, return FALSE, store for 1 hour */
  53.         if ( 'error' == $calibrefx_update || is_wp_error( $calibrefx_update ) || ! is_serialized( $calibrefx_update ) ) {
  54.             set_transient( 'calibrefx-update', array( 'new_version' => FRAMEWORK_VERSION ), 60 * 60 );
  55.             return false;
  56.         }
  57.  
  58.         /** Else, unserialize */
  59.         $calibrefx_update = maybe_unserialize( $calibrefx_update );
  60.  
  61.         /** And store in transient for 24 hours */
  62.         set_transient( 'calibrefx-update', $calibrefx_update, 60 * 60 * 24 );
  63.     }
  64.     
  65.     /** If we're already using the latest version, return false */
  66.     if ( version_compare( FRAMEWORK_VERSION, $calibrefx_update['new_version'], '>=' ) )
  67.         return false;
  68.  
  69.     return $calibrefx_update;
  70. }
