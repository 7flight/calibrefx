  78. function calibrefx_admin_bar_my_account_item( $calibrefx_admin_bar ) {
  79.     $user_id      = get_current_user_id();

  81.     $profile_url  = get_edit_profile_url( $user_id );
  82.  
  83.     if ( ! $user_id )

  94.         'href'      => $profile_url,
  95.         'meta'      => array(
  96.             'class'     => $class,
  97.             'title'     => __('My Account'),
  98.         ),
  99.     ) );

  206. function calibrefx_admin_bar_site_menu( $calibrefx_admin_bar ) {
  207.     global $current_site;
  208.  
  209.     // Don't show for logged out users.
  210.     if ( ! is_user_logged_in() )
  211.         return;
  212.  
  213.     // Show only when the user is a member of this site, or they're a super admin.
  214.     if ( ! is_user_member_of_blog() && ! is_super_admin() )
  215.         return;
  216.  
  217.     $blogname = get_bloginfo('name');
  218.  
  219.     if ( empty( $blogname ) )
  220.         $blogname = preg_replace( '#^(https?://)?(www.)?#', '', get_home_url() );
  221.  
  222.     if ( is_network_admin() ) {
  223.         $blogname = sprintf( __('Network Admin: %s'), esc_html( $current_site->site_name ) );
  224.     } elseif ( is_user_admin() ) {
  225.         $blogname = sprintf( __('Global Dashboard: %s'), esc_html( $current_site->site_name ) );
  226.     }
  227.  
  228.     $title = wp_html_excerpt( $blogname, 40 );
  229.     if ( $title != $blogname )
  230.         $title = trim( $title ) . '&hellip;';
  231.  
  232.     $calibrefx_admin_bar->add_menu( array(
  233.         'id'    => 'site-name',
  234.         'title' => $title,
  235.         'href'  => is_admin() ? home_url( '/' ) : admin_url(),
  236.     ) );
  237.  
  238.     // Create submenu items.
  239.  
  240.     if ( is_admin() ) {
  241.         // Add an option to visit the site.
  242.         $calibrefx_admin_bar->add_menu( array(
  243.             'parent' => 'site-name',
  244.             'id'     => 'view-site',
  245.             'title'  => __( 'Visit Site' ),
  246.             'href'   => home_url( '/' ),
  247.         ) );
  248.  
  249.     // We're on the front end, print a copy of the admin menu.
  250.     } else {
  251.         // Add the dashboard item.
  252.         $calibrefx_admin_bar->add_menu( array(
  253.             'parent' => 'site-name',
  254.             'id'     => 'dashboard',
  255.             'title'  => __( 'Dashboard' ),
  256.             'href'   => admin_url(),
  257.         ) );
  258.  
  259.         // Add the appearance submenu items.
  260.         calibrefx_admin_bar_appearance_menu( $calibrefx_admin_bar );
  261.     }
