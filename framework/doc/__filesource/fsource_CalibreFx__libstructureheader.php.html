  204. /**
  205.  * This function will activate the custom header image from WordPress
  206.  *
  207.  * It gets arguments passed through add_theme_support(), defines the constants,
  208.  * and calls add_custom_image_header().
  209.  *
  210.  * @return void 
  211.  */
  212. function calibrefx_custom_header() {
  213.  
  214.     $custom_header = get_theme_support( 'calibrefx-custom-header' );
  215.  
  216.     /** If not active, do nothing */
  217.     if ( ! $custom_header )
  218.         return;
  219.  
  220.     /** Blog title option is obsolete when custom header is active */
  221.     add_filter( 'calibrefx_pre_get_option_blog_title', '__return_empty_array' );
  222.  
  223.     /** Cast, if necessary */
  224.     $custom_header = isset( $custom_header[0] ) && is_array( $custom_header[0] ) ? $custom_header[0] : array( );
  225.  
  226.     /** Merge defaults with passed arguments */
  227.     $args = wp_parse_args( $custom_header, array(
  228.         'width'                 => 960,
  229.         'height'                => 80,
  230.         'textcolor'             => '333333',
  231.         'no_header_text'        => false,
  232.         'header_image'          => '%s/header.png',
  233.         'header_callback'       => 'calibrefx_custom_header_style',
  234.         'admin_header_callback' => 'calibrefx_custom_header_admin_style'
  235.     ) );
  236.  
  237.     /** Define all the constants */
  238.     if ( !defined( 'HEADER_IMAGE_WIDTH' ) && is_numeric( $args['width'] ) )
  239.         define( 'HEADER_IMAGE_WIDTH', $args['width'] );
  240.  
  241.     if ( !defined( 'HEADER_IMAGE_HEIGHT' ) && is_numeric( $args['height'] ) )
  242.         define( 'HEADER_IMAGE_HEIGHT', $args['height'] );
  243.  
  244.     if ( !defined( 'HEADER_TEXTCOLOR' ) && $args['textcolor'] )
  245.         define( 'HEADER_TEXTCOLOR', $args['textcolor'] );
  246.  
  247.     if ( !defined( 'NO_HEADER_TEXT' ) && $args['no_header_text'] )
  248.         define( 'NO_HEADER_TEXT', $args['no_header_text'] );
  249.  
  250.     if ( !defined( 'HEADER_IMAGE' ) && $args['header_image'] )
  251.         define( 'HEADER_IMAGE', sprintf( $args['header_image'], apply_filters('calibrefx_images_url', CALIBREFX_IMAGES_URL) ) );
  252.  
  253.     /** Activate Custom Header */
  254.     add_custom_image_header( $args['header_callback'], $args['admin_header_callback'] );
  255.  
  256. }

  258. /**
  259.  * Header callback. It outputs special CSS to the document
  260.  * head, modifying the look of the header based on user input.
  261.  *
  262.  */
  263. function calibrefx_custom_header_style() {
  264.     /** If no options set, don't waste the output. Do nothing. */
  265.     if ( HEADER_TEXTCOLOR == get_header_textcolor() && HEADER_IMAGE == get_header_image() )
  266.         return;
  267.     
  268.     $header = sprintf( '#header { background: url(%s) no-repeat; }', esc_url( get_header_image() ) );
  269.     $text = sprintf( '#title a, #title a:hover, #description { color: #%s; }', esc_html( get_header_textcolor() ) );
  270.  
  271.     printf( '<style type="text/css">%1$s %2$s</style>', $header, $text );
  272.  
  273. }
  274.  
  275. /**
  276.  * Header admin callback. It outputs special CSS to the admin
  277.  * document head, modifying the look of the header area based on user input.
  278.  *
  279.  * Will probably need to be overridden in the child theme with a custom callback.
  280.  */
  281. function calibrefx_custom_header_admin_style() {
  282.  
  283.     $headimg = sprintf( '.appearance_page_custom-header #headimg { background: url(%s) no-repeat; min-height: %spx; }', get_header_image(), HEADER_IMAGE_HEIGHT );
  284.     $h1 = sprintf( '#headimg h1, #headimg h1 a { color: #%s; font-size: 24px; font-weight: normal; line-height: 30px; margin: 20px 0 0; text-decoration: none; }', esc_html( get_header_textcolor() ) );
  285.     $desc = sprintf( '#headimg #desc { color: #%s; font-size: 12px; font-style: italic; line-height: 1; margin: 0; }', esc_html( get_header_textcolor() ) );
  286.  
  287.     printf( '<style type="text/css">%1$s %2$s %3$s</style>', $headimg, $h1, $desc );
  288.  
  289. }
