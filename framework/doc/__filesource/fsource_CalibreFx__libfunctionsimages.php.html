  47. /**
  48.  * Get an image from the media gallery and returns it
  49.  */
  50. function calibrefx_get_image($args = array()) {
  51.     global $post;
  52.  
  53.     $defaults = array(
  54.         'format' => 'html',
  55.         'size' => 'full',
  56.         'num' => 0,
  57.         'attr' => ''
  58.     );
  59.     $defaults = apply_filters('calibrefx_get_image_default_args', $defaults);
  60.  
  61.     $args = wp_parse_args($args, $defaults);
  62.  
  63.     // Allow child theme to short-circuit this function
  64.     $pre = apply_filters('calibrefx_pre_get_image', false, $args, $post);
  65.     if ( false !== $pre ) return $pre;
  66.  
  67.     // check for feature image
  68.     if ( has_post_thumbnail() && ($args['num'] === 0) ) {
  69.         $id = get_post_thumbnail_id();
  70.         $html = wp_get_attachment_image($id, $args['size'], false, $args['attr']);
  71.         list($url) = wp_get_attachment_image_src($id, $args['size'], false, $args['attr']);
  72.     }
  73.     else {
  74.         $id = calibrefx_get_image_id($args['num']);
  75.         $html = wp_get_attachment_image($id, $args['size'], false, $args['attr']);
  76.         list($url) = wp_get_attachment_image_src($id, $args['size'], false, $args['attr']);
  77.     }
  78.  
  79.     // source path, relative to the root
  80.     $src = str_replace( home_url(), '', $url );
  81.  
  82.     // determine output
  83.     if ( strtolower($args['format']) == 'html' )
  84.         $output = $html;
  85.     elseif ( strtolower($args['format']) == 'url' )
  86.         $output = $url;
  87.     else
  88.         $output = $src;
  89.  
  90.     // $url is blank return false
  91.     if ( empty($url) ) $output = FALSE;
  92.  
  93.     // return data, filtered
  94.     return apply_filters('calibrefx_get_image', $output, $args, $id, $html, $url, $src);
  95. }
