  31. function calibrefx_register_sidebar($args) {
  32.     $defaults = (array) apply_filters( 'calibrefx_register_sidebar_defaults', array(
  33.         'before_widget' => '<div id="%1$s" class="widget %2$s"><div class="widget-wrap">',
  34.         'after_widget'  => "</div></div>\n",
  35.         'before_title'  => '<h4 class="widgettitle">',
  36.         'after_title'   => "</h4>\n"
  37.     ) );
  38.  
  39.     $args = wp_parse_args($args, $defaults);
  40.  
  41.     return register_sidebar($args);
  42. }
